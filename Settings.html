// --- 2. 데이터 정제 및 DB 저장 진행 --- function handleDataProcessing(rawData) { const resultDiv = document.getElementById('roster-result'); const rows = rawData.split('\n'); let successCount = 0; let failCount = 0; let updates = {}; let firstErrorMessage = ""; rows.forEach((row, index) => { if (!row.trim()) return; // 완전 빈 줄은 그냥 넘어갑니다. const cols = row.split(/[\t, ]+/).filter(Boolean); // 🌟 핵심 추가 코드: 첫 번째 줄(index 0)의 첫 칸이 숫자가 아니면(즉, '학년' 같은 제목이면) 건너뜁니다. if (index === 0 && isNaN(cols[0])) { return; } // [검사 1] 항목 갯수 확인 (최소 4개 필요) if (cols.length < 4) { failCount++; if (!firstErrorMessage) { firstErrorMessage = `${index + 1}번째 줄 오류: 항목이 ${cols.length}개뿐입니다.
➔ 데이터: "${row.substring(0, 20)}..."`; } return; } const grade = cols[0].trim(); const classNum = cols[1].trim(); const studentNum = cols[2].trim(); // [검사 2] 학년, 반, 번호가 숫자인지 확인 if (isNaN(grade) || isNaN(classNum) || isNaN(studentNum)) { failCount++; if (!firstErrorMessage) { firstErrorMessage = `${index + 1}번째 줄 오류: 학년/반/번호는 '숫자'여야 합니다.
➔ 데이터: "${row.substring(0, 20)}..."`; } return; } // 정상 데이터 처리 const cleanName = cols[3].replace(/[\s!@#$%^&*(),.?":{}|<>]/g, ''); const phone = cols[4] ? cols[4].trim() : "없음"; const studentId = `${grade}${classNum.padStart(2, '0')}${studentNum.padStart(2, '0')}`; updates['settings/roster/' + studentId] = { name: cleanName, phone: phone, grade: grade }; successCount++; }); // 🔴 완전 실패 시 if (successCount === 0) { resultDiv.style.color = "red"; resultDiv.innerHTML = `❌ 저장 실패: 올바른 형식이 아닙니다.

${firstErrorMessage || '데이터가 비어있습니다.'}`; alert("데이터 변환에 실패했습니다. 상세 원인을 확인해주세요."); return; } // 🟡 일부 실패 경고 메시지 let warningMsg = ""; if (failCount > 0) { warningMsg = `

⚠️ 주의: ${failCount}개의 줄은 형식이 맞지 않아 제외되었습니다.
(예: ${firstErrorMessage})`; } // [상황 표시] resultDiv.style.color = "blue"; resultDiv.innerText = `⏳ 총 ${successCount}명의 데이터를 서버에 저장하는 중입니다...`; // Firebase 전송 db.ref().update(updates).then(() => { resultDiv.style.color = "green"; resultDiv.innerHTML = `✅ 성공! 총 ${successCount}명의 명단이 시스템에 등록되었습니다.${warningMsg}`; alert(`[등록 완료] ${successCount}명 저장 성공! ${failCount > 0 ? `(제외된 데이터: ${failCount}건)` : ''}`); document.getElementById('roster-input').value = ''; if(document.getElementById('roster-file')) document.getElementById('roster-file').value = ''; }).catch(error => { resultDiv.style.color = "red"; resultDiv.innerText = "❌ 서버 저장 실패: " + error.message; alert("저장 중 서버 오류가 발생했습니다."); }); }
